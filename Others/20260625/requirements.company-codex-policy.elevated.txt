# =====================================================================
# requirements.toml - Company Codex managed requirements
# =====================================================================
# ChatGPT Codex > Cloud managed requirements に貼り付ける前提。
#
# 目的:
# - requirements.toml のみでユーザーが上書きできない制約をかける。
# - Codex 0.138.0+ では permission profile で Workspace 内だけ編集可にする。
# - 旧クライアント向けには allowed_sandbox_modes で full access を遮断する。
# =====================================================================

# ---------------------------------------------------------------------
# 1) 承認ポリシー
# ---------------------------------------------------------------------

# 「never」を許可しない。untrusted command は承認ゲートを通す。
allowed_approval_policies = ["untrusted"]

# 承認者をユーザーに固定する。auto_review による自動承認へ逃がさない。
allowed_approvals_reviewers = ["user"]

# ---------------------------------------------------------------------
# 2) 旧クライアント互換: sandbox_mode
# ---------------------------------------------------------------------

# danger-full-access / --yolo を禁止する。
# 旧クライアントでは permission profile を無視するため、この行で最低限の互換制御を行う。
allowed_sandbox_modes = ["read-only", "workspace-write"]

# ---------------------------------------------------------------------
# 3) Web 検索
# ---------------------------------------------------------------------

# live Web 取得を禁止し、OpenAI 管理インデックスの cached 検索だけ許可する。
# 仕様上 "disabled" は常に許可されるため、実際には「cached または disabled」になる。
allowed_web_search_modes = ["cached"]


# ---------------------------------------------------------------------
# 4) Codex 0.138.0+ 向け: 既定 permission profile
# ---------------------------------------------------------------------

# ユーザーの既定 profile を管理者定義の Workspace 専用 profile に固定する。
# 重要: TOML では [table] の後に書いたキーはその table 配下になるため、
# default_permissions は必ず最初の [table] より前、トップレベルに置く。
default_permissions = "company_workspace_only"

# ---------------------------------------------------------------------
# 5) Hooks / MCP / Plugin / App / Browser / Computer Use
# ---------------------------------------------------------------------

# ユーザー、プロジェクト、セッション、プラグイン由来の hooks を読み込ませない。
# 管理者が managed layer で定義した hooks だけ許可する。
allow_managed_hooks_only = true

# 端末のリモートコントロールを禁止する。
allow_remote_control = false

# Appshots はコード・画面情報の漏えい面を増やすため無効化する。
allow_appshots = false

[features]
# 管理者管理 hooks を有効化する。allow_managed_hooks_only=true と併用する。
hooks = true

# permission profile の外側の操作面を減らす。
apps = false
browser_use = false
browser_use_external = false
browser_use_full_cdp_access = false
in_app_browser = false
computer_use = false

# 複数 agent による追加実行面を使わせない。
multi_agent = false

# ユーザー導入プラグインとプラグイン共有を止める。
plugins = false
plugin_sharing = false

[computer_use]
# 万一 Computer Use が有効なクライアントでも、管理 macOS ロック後の操作を禁止する。
allow_locked_computer_use = false

# MCP は管理者 allowlist 方式に限定する。
# 空の [mcp_servers] は、管理者が明示していない MCP server をすべて無効化する。
[mcp_servers]

# 社内 MCP を許可する場合だけ、管理者が次のような allowlist を追加する。
# [mcp_servers.internal_docs.identity]
# url = { match = "exact", value = "https://mcp.example.co.jp/codex" }

[marketplaces]
# ユーザーが任意の marketplace source から plugin を追加・インストール・更新することを禁止する。
restrict_to_allowed_sources = true

# ---------------------------------------------------------------------
# 6) shell / subprocess 側のネットワーク遮断
# ---------------------------------------------------------------------

[experimental_network]
# sandboxed networking requirements を有効化する。
# これ自体は command network access を付与しない。
enabled = true

# 管理者定義の allowlist だけを有効にする。
# allowed_domains を空にしているため、ユーザー側の許可追加は効かない。

allowed_domains = []

# ローカル・プライベートネットワークや上流プロキシ経由の抜け道を閉じる。
allow_local_binding = false
allow_upstream_proxy = false

dangerously_allow_non_loopback_proxy = false

  # OS 権限昇格は禁止する。絶対パスや Windows の runas/Start-Process 経由も明示的に拒否する。
  { pattern = [{ any_of = ["sudo", "/usr/bin/sudo", "/bin/sudo", "su", "/usr/bin/su", "/bin/su", "doas", "/usr/bin/doas", "/bin/doas", "pkexec", "/usr/bin/pkexec", "/bin/pkexec", "runas", "runas.exe", "Start-Process", "start-process"] }], decision = "forbidden", justification = "Company policy: OS privilege escalation is not allowed from Codex." },

  # 共有フォルダ・外部ボリューム・FUSE/SSHFS を後付け mount する経路を禁止する。
  # Windows では drive mapping / UNC share 作成 / ボリューム操作につながる代表コマンドも拒否する。
  { pattern = [{ any_of = ["mount", "umount", "sshfs", "fusermount", "fusermount3", "diskutil", "mountvol", "mountvol.exe", "diskpart", "diskpart.exe", "subst", "subst.exe", "New-PSDrive", "new-psdrive", "New-SmbMapping", "new-smbmapping", "Remove-SmbMapping", "remove-smbmapping"] }], decision = "forbidden", justification = "Company policy: mounting, mapping, or unmounting shared/external storage is not allowed from Codex." },

  # Windows: net use/share/user/localgroup など、共有・アカウント・サービス状態に影響する net 操作を禁止する。
  { pattern = [{ any_of = ["net", "net.exe"] }, { any_of = ["use", "USE", "Use", "share", "SHARE", "Share", "user", "USER", "User", "localgroup", "LOCALGROUP", "Localgroup", "group", "GROUP", "Group", "session", "SESSION", "Session", "file", "FILE", "File", "start", "START", "Start", "stop", "STOP", "Stop"] }], decision = "forbidden", justification = "Company policy: Windows network-share, account, and service-control net commands are not allowed from Codex." },

  # Windows: native Windows から WSL を起動して /mnt 経由でホスト側へ抜ける経路を禁止する。
  { pattern = [{ any_of = ["wsl", "wsl.exe"] }], decision = "forbidden", justification = "Company policy: launching WSL from native Windows Codex is not allowed because it can expose mounted Windows drives." },

  # コンテナ・クラスタ・IaC はホスト/クラウド側へ影響しやすいため禁止する。
  { pattern = [{ any_of = ["docker", "podman", "kubectl", "helm", "terraform", "tofu"] }], decision = "forbidden", justification = "Company policy: tools that can affect host, cluster, or cloud resources are not allowed from Codex." },

  # シェルや実行系・パッケージ管理系は承認ゲートを通す。
  # Windows の cmd / PowerShell / .NET / Windows package manager も含める。
  { pattern = [{ any_of = ["bash", "sh", "zsh", "fish", "python", "python3", "py", "py.exe", "node", "npm", "npx", "pnpm", "yarn", "bun", "pip", "pip3", "uv", "make", "cmake", "cargo", "go", "mvn", "gradle", "cmd", "cmd.exe", "powershell", "powershell.exe", "pwsh", "pwsh.exe", "dotnet", "msbuild", "msbuild.exe", "winget", "winget.exe", "choco", "choco.exe", "scoop", "scoop.cmd", "msiexec", "msiexec.exe"] }], decision = "prompt", justification = "Company policy: executable or package/build commands require explicit approval." },
